package com.pharmacy.views;

import com.pharmacy.models.Medicine;

import javax.swing.*;
import javax.swing.border.EmptyBorder;
import java.awt.*;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;
import java.util.function.Consumer;

import static com.pharmacy.views.ThemeConstants.*;

public class MedicineCard extends JPanel {

    /** How this card should be visually styled. */
    public enum ExpiryMode {
        NORMAL,         // default: orange left stripe
        EXPIRING_SOON,  // red outline + ⚠ badge
        EXPIRY_SAFE     // green outline (closest but not urgent)
    }

    private final Medicine medicine;
    private final ExpiryMode mode;
    private boolean hovered = false;

    // Convenience constructor — normal mode
    public MedicineCard(Medicine medicine, String brandName, String presTypeName, Consumer<Medicine> stockUpdater) {
        this(medicine, brandName, presTypeName, ExpiryMode.NORMAL, stockUpdater);
    }

    public MedicineCard(Medicine medicine, String brandName, String presTypeName, ExpiryMode mode, Consumer<Medicine> stockUpdater) {
        this.medicine = medicine;
        this.mode = mode;

        setPreferredSize(new Dimension(240, 160));
        setMinimumSize(new Dimension(200, 145));
        setLayout(new BorderLayout());
        setOpaque(false);
        setCursor(new Cursor(Cursor.HAND_CURSOR));

        addMouseListener(new MouseAdapter() {
            @Override public void mouseEntered(MouseEvent e) { hovered = true;  repaint(); }
            @Override public void mouseExited (MouseEvent e) { hovered = false; repaint(); }
        });

        // ── Content ───────────────────────────────────────────────────────────
        JPanel content = new JPanel();
        content.setLayout(new BoxLayout(content, BoxLayout.Y_AXIS));
        content.setOpaque(false);
        // Left padding is larger for EXPIRING_SOON (no stripe, we use border outline instead)
        int leftPad = (mode == ExpiryMode.NORMAL) ? 12 : 14;
        content.setBorder(new EmptyBorder(12, leftPad, 12, 12));

        JLabel nameLabel = new JLabel(medicine.getMedName());
        nameLabel.setFont(FONT_HEADER);
        nameLabel.setForeground(TEXT_PRIMARY);
        nameLabel.setAlignmentX(Component.LEFT_ALIGNMENT);

        JLabel presLabel = new JLabel(presTypeName != null ? presTypeName : "Unknown");
        presLabel.setFont(FONT_SMALL);
        presLabel.setForeground(Color.WHITE);
        presLabel.setBackground(mode == ExpiryMode.EXPIRING_SOON ? new Color(210, 60, 60) : ACCENT);
        presLabel.setOpaque(true);
        presLabel.setBorder(new EmptyBorder(2, 7, 2, 7));

        JLabel brandLabel = new JLabel("Brand: " + (brandName != null ? brandName : "Unknown"));
        brandLabel.setFont(FONT_SMALL);
        brandLabel.setForeground(TEXT_SECONDARY);
        brandLabel.setAlignmentX(Component.LEFT_ALIGNMENT);

        JLabel qtyLabel = new JLabel("Stock: " + medicine.getQuantity() + " units");
        qtyLabel.setFont(new Font("SansSerif", Font.BOLD, 12));
        qtyLabel.setForeground(medicine.getQuantity() > 10 ? new Color(50, 160, 90) : new Color(210, 70, 70));

        // Show expiration date on expiry-mode cards
        JLabel expLabel = null;
        if (mode != ExpiryMode.NORMAL && medicine.getExpirationDate() != null) {
            expLabel = new JLabel("Exp: " + medicine.getExpirationDate());
            expLabel.setFont(new Font("SansSerif", Font.BOLD, 11));
            expLabel.setForeground(mode == ExpiryMode.EXPIRING_SOON
                    ? new Color(200, 50, 50) : new Color(50, 150, 80));
        }

        JPanel qtyRow = new JPanel(new BorderLayout());
        qtyRow.setOpaque(false);
        qtyRow.setAlignmentX(Component.LEFT_ALIGNMENT);
        qtyRow.setMaximumSize(new Dimension(Short.MAX_VALUE, 40));

        JPanel labelsPanel = new JPanel();
        labelsPanel.setLayout(new BoxLayout(labelsPanel, BoxLayout.Y_AXIS));
        labelsPanel.setOpaque(false);
        labelsPanel.add(qtyLabel);
        if (expLabel != null) {
            labelsPanel.add(Box.createVerticalStrut(3));
            labelsPanel.add(expLabel);
        }

        JButton decBtn = new CircleButton("-");
        JButton incBtn = new CircleButton("+");

        if (stockUpdater != null) {
            decBtn.addActionListener(e -> {
                if (medicine.getQuantity() > 0) {
                    medicine.setQuantity(medicine.getQuantity() - 1);
                    stockUpdater.accept(medicine);
                    qtyLabel.setText("Stock: " + medicine.getQuantity() + " units");
                    qtyLabel.setForeground(medicine.getQuantity() > 10 ? new Color(50, 160, 90) : new Color(210, 70, 70));
                    repaint();
                }
            });
            incBtn.addActionListener(e -> {
                medicine.setQuantity(medicine.getQuantity() + 1);
                stockUpdater.accept(medicine);
                qtyLabel.setText("Stock: " + medicine.getQuantity() + " units");
                qtyLabel.setForeground(medicine.getQuantity() > 10 ? new Color(50, 160, 90) : new Color(210, 70, 70));
                repaint();
            });
        } else {
            decBtn.setEnabled(false);
            incBtn.setEnabled(false);
            decBtn.setVisible(false);
            incBtn.setVisible(false);
        }

        JPanel btnsPanel = new JPanel();
        btnsPanel.setLayout(new BoxLayout(btnsPanel, BoxLayout.X_AXIS));
        btnsPanel.setOpaque(false);
        btnsPanel.add(decBtn);
        btnsPanel.add(Box.createHorizontalStrut(5));
        btnsPanel.add(incBtn);

        qtyRow.add(labelsPanel, BorderLayout.CENTER);
        qtyRow.add(btnsPanel, BorderLayout.EAST);

        JPanel badgeRow = new JPanel(new FlowLayout(FlowLayout.LEFT, 0, 0));
        badgeRow.setOpaque(false);
        badgeRow.setAlignmentX(Component.LEFT_ALIGNMENT);
        badgeRow.add(presLabel);

        content.add(nameLabel);
        content.add(Box.createVerticalStrut(5));
        content.add(badgeRow);
        content.add(Box.createVerticalStrut(5));
        content.add(brandLabel);
        content.add(Box.createVerticalStrut(3));
        content.add(qtyRow);

        JPanel inner = new JPanel(new BorderLayout());
        inner.setOpaque(false);
        // Only normal mode has the left stripe offset
        if (mode == ExpiryMode.NORMAL) {
            inner.setBorder(new EmptyBorder(0, 6, 0, 0));
        }
        inner.add(content, BorderLayout.CENTER);
        add(inner, BorderLayout.CENTER);
    }

    @Override
    protected void paintComponent(Graphics g) {
        Graphics2D g2 = (Graphics2D) g.create();
        g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);

        int w = getWidth(), h = getHeight();
        int arc = 14;
        int shadowOffset = hovered ? 4 : 2;
        int alpha = hovered ? 30 : 18;

        // Shadow
        g2.setColor(new Color(0, 0, 0, alpha));
        g2.fillRoundRect(shadowOffset, shadowOffset, w - shadowOffset, h - shadowOffset, arc, arc);

        // Card background
        g2.setColor(BG_CARD);
        g2.fillRoundRect(0, 0, w - shadowOffset, h - shadowOffset, arc, arc);

        if (mode == ExpiryMode.NORMAL) {
            // Orange left stripe
            Color stripe = hovered ? ACCENT_HOVER : ACCENT;
            g2.setColor(stripe);
            g2.fillRoundRect(0, 0, 10, h - shadowOffset, arc, arc);
            g2.fillRect(6, 0, 4, h - shadowOffset);

        } else if (mode == ExpiryMode.EXPIRING_SOON) {
            // Red outline (3px)
            g2.setColor(new Color(210, 55, 55));
            g2.setStroke(new BasicStroke(3f));
            g2.drawRoundRect(1, 1, w - shadowOffset - 2, h - shadowOffset - 2, arc, arc);

            // ⚠ badge top-right
            g2.setFont(new Font("SansSerif", Font.BOLD, 18));
            g2.setColor(new Color(210, 55, 55));
            FontMetrics fm = g2.getFontMetrics();
            String warn = "⚠";
            g2.drawString(warn, w - shadowOffset - fm.stringWidth(warn) - 8, 22);

        } else { // EXPIRY_SAFE — green outline
            g2.setColor(new Color(50, 160, 90));
            g2.setStroke(new BasicStroke(3f));
            g2.drawRoundRect(1, 1, w - shadowOffset - 2, h - shadowOffset - 2, arc, arc);
        }

        g2.dispose();
    }

    public Medicine getMedicine() { return medicine; }

    static class CircleButton extends JButton {
        public CircleButton(String text) {
            super(text);
            setContentAreaFilled(false);
            setFocusPainted(false);
            setBorderPainted(false);
            setOpaque(false);
            setCursor(new Cursor(Cursor.HAND_CURSOR));
            setPreferredSize(new Dimension(30, 30));
            setMinimumSize(new Dimension(30, 30));
            setMaximumSize(new Dimension(30, 30));
        }
        @Override protected void paintComponent(Graphics g) {
            Graphics2D g2 = (Graphics2D) g.create();
            g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
            if (getModel().isRollover() || getModel().isArmed()) {
                g2.setColor(ACCENT_LIGHT);
                g2.fillOval(1, 1, getWidth()-3, getHeight()-3);
            }
            g2.setColor(ACCENT);
            g2.setStroke(new BasicStroke(2f));
            g2.drawOval(1, 1, getWidth()-3, getHeight()-3);
            g2.setFont(new Font("SansSerif", Font.BOLD, 16));
            g2.setColor(ACCENT);
            FontMetrics fm = g2.getFontMetrics();
            String label = getText();
            int tx = (getWidth() - fm.stringWidth(label)) / 2;
            int ty = (getHeight() - fm.getHeight()) / 2 + fm.getAscent();
            g2.drawString(label, tx, ty);
            g2.dispose();
        }
    }
}
